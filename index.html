package com.example.attendancescanner; import android.Manifest; import android.content.pm.PackageManager; import android.graphics.Color; import android.os.Bundle; import android.view.View; import android.webkit.PermissionRequest; import android.webkit.WebChromeClient; import android.webkit.WebSettings; import android.webkit.WebView; import android.webkit.WebViewClient; import android.widget.Button; import android.widget.EditText; import android.widget.LinearLayout; import androidx.appcompat.app.AppCompatActivity; import androidx.core.app.ActivityCompat; import androidx.core.content.ContextCompat; import com.google.android.material.bottomnavigation.BottomNavigationView; public class MainActivity extends AppCompatActivity { private WebView myWebView; @Override protected void onCreate(Bundle savedInstanceState) { super.onCreate(savedInstanceState); setContentView(R.layout.activity_main); myWebView = findViewById(R.id.webview); EditText manualInput = findViewById(R.id.manualInput); Button btnSubmitManual = findViewById(R.id.btnSubmitManual); Button btnIn = findViewById(R.id.btnIn); Button btnOut = findViewById(R.id.btnOut); LinearLayout controls = findViewById(R.id.controlsLayout); BottomNavigationView bottomNav = findViewById(R.id.bottomNav); // Styling for the Professional Blue Look btnSubmitManual.setBackgroundColor(Color.parseColor("#1e40af")); // Navy Blue btnIn.setBackgroundColor(Color.parseColor("#3b82f6")); // Bright Blue btnOut.setBackgroundColor(Color.parseColor("#94a3b8")); // Slate Grey (Inactive) WebSettings webSettings = myWebView.getSettings(); webSettings.setJavaScriptEnabled(true); webSettings.setDomStorageEnabled(true); myWebView.setWebViewClient(new WebViewClient()); myWebView.setWebChromeClient(new WebChromeClient() { @Override public void onPermissionRequest(final PermissionRequest request) { request.grant(request.getResources()); } }); myWebView.loadUrl("https://everloungei-boop.github.io/qr-attendance/"); bottomNav.setOnItemSelectedListener(item -> { String page; int id = item.getItemId(); if (id == R.id.nav_scan) { page = "scan"; controls.setVisibility(View.VISIBLE); } else { page = (id == R.id.nav_history) ? "history" : (id == R.id.nav_users) ? "users" : "settings"; controls.setVisibility(View.GONE); } myWebView.evaluateJavascript("javascript:navigate('" + page + "')", null); return true; }); btnIn.setOnClickListener(v -> { myWebView.evaluateJavascript("javascript:setMode('IN')", null); btnIn.setBackgroundColor(Color.parseColor("#3b82f6")); // Active Blue btnOut.setBackgroundColor(Color.parseColor("#94a3b8")); // Inactive Grey }); btnOut.setOnClickListener(v -> { myWebView.evaluateJavascript("javascript:setMode('OUT')", null); btnOut.setBackgroundColor(Color.parseColor("#1e40af")); // Active Navy btnIn.setBackgroundColor(Color.parseColor("#94a3b8")); // Inactive Grey }); btnSubmitManual.setOnClickListener(v -> { String text = manualInput.getText().toString().trim(); if (!text.isEmpty()) { myWebView.evaluateJavascript("javascript:handleManualScan('" + text + "')", null); manualInput.setText(""); } }); if (ContextCompat.checkSelfPermission(this, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) { ActivityCompat.requestPermissions(this, new String[]{Manifest.permission.CAMERA}, 1); } } }
